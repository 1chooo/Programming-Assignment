```
[['2 * 2 = 4', '2 * 3 = 6', '2 * 4 = 8', ... , '2 * 9 = 18'],
['3 * 2 = 6', '3 * 3 = 9', '3 * 4 = 12', ... , '3 * 9 = 27'],
```
